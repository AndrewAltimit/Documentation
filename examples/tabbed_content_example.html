Tabbed Content for Technical Documentation

This example demonstrates how to implement tabbed content in documentation pages. This is particularly useful for showing code examples in multiple languages, different approaches to solving a problem, or organizing complex information.

Example 1: Multi-Language Code Examples

Show the same algorithm implemented in different programming languages:

Fibonacci Implementation in Python

def fibonacci(n):
    """Calculate the nth Fibonacci number using dynamic programming"""
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    
    # Use dynamic programming for efficiency
    fib = [0, 1]
    for i in range(2, n + 1):
        fib.append(fib[i-1] + fib[i-2])
    
    return fib[n]

# Example usage
for i in range(10):
    print(f"F({i}) = {fibonacci(i)}")

Fibonacci Implementation in JavaScript

function fibonacci(n) {
    // Calculate the nth Fibonacci number using dynamic programming
    if (n <= 0) return 0;
    if (n === 1) return 1;
    
    // Use dynamic programming for efficiency
    const fib = [0, 1];
    for (let i = 2; i <= n; i++) {
        fib[i] = fib[i-1] + fib[i-2];
    }
    
    return fib[n];
}

// Example usage
for (let i = 0; i < 10; i++) {
    console.log(`F(${i}) = ${fibonacci(i)}`);
}

Fibonacci Implementation in Java

public class Fibonacci {
    public static int fibonacci(int n) {
        // Calculate the nth Fibonacci number using dynamic programming
        if (n <= 0) return 0;
        if (n == 1) return 1;
        
        // Use dynamic programming for efficiency
        int[] fib = new int[n + 1];
        fib[0] = 0;
        fib[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            fib[i] = fib[i-1] + fib[i-2];
        }
        
        return fib[n];
    }
    
    public static void main(String[] args) {
        for (int i = 0; i < 10; i++) {
            System.out.println("F(" + i + ") = " + fibonacci(i));
        }
    }
}

Fibonacci Implementation in Rust

fn fibonacci(n: u32) -> u64 {
    // Calculate the nth Fibonacci number using dynamic programming
    match n {
        0 => 0,
        1 => 1,
        _ => {
            let mut fib = vec![0u64, 1];
            for i in 2..=n as usize {
                let next = fib[i-1] + fib[i-2];
                fib.push(next);
            }
            fib[n as usize]
        }
    }
}

fn main() {
    for i in 0..10 {
        println!("F({}) = {}", i, fibonacci(i));
    }
}

Example 3: Different Approaches to a Problem

Compare different algorithmic approaches with their trade-offs:

Recursive Approach

Time Complexity: O(2^n) | Space Complexity: O(n)

Simple but inefficient for large values due to repeated calculations.

def fib_recursive(n):
    if n <= 1:
        return n
    return fib_recursive(n-1) + fib_recursive(n-2)

Iterative Approach

Time Complexity: O(n) | Space Complexity: O(1)

More efficient than recursion, uses constant space.

def fib_iterative(n):
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

Memoized Approach

Time Complexity: O(n) | Space Complexity: O(n)

Combines recursion with caching to avoid repeated calculations.

def fib_memoized(n, memo={}):
    if n in memo:
        return memo[n]
    if n <= 1:
        return n
    memo[n] = fib_memoized(n-1, memo) + fib_memoized(n-2, memo)
    return memo[n]

Tabulated Approach

Time Complexity: O(n) | Space Complexity: O(n)

Bottom-up dynamic programming approach.

def fib_tabulated(n):
    if n <= 1:
        return n
    dp = [0] * (n + 1)
    dp[1] = 1
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]
